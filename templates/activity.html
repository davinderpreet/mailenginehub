Tracking Pixel — Install on Your Shopify Store
Paste this snippet into your Shopify theme's theme.liquid just before </body> to track real-time product views and page visits.
<!-- MailEngineHub Tracking Pixel v2 --> <!-- Step 1: Add this ABOVE the pixel in theme.liquid (identifies logged-in customers): --> <!-- {% if customer %}<script>window.meh_email='{{ customer.email }}';</script>{% endif %} --> <script> (function() { var ENDPOINT = '{{ site_url }}/api/track'; var sid = localStorage.getItem('meh_sid') || Math.random().toString(36).substr(2,9); localStorage.setItem('meh_sid', sid); // Email ID: Liquid injection > URL param ?meh_e= > localStorage > anonymous var email = window.meh_email || ''; if (!email) { var urlEmail = new URLSearchParams(window.location.search).get('meh_e'); if (urlEmail) { email = urlEmail; localStorage.setItem('meh_email', decodeURIComponent(urlEmail)); } } if (!email) email = localStorage.getItem('meh_email') || ''; function track(type, data) { var payload = { event_type: type, event_data: data, session_id: sid }; if (email) payload.email = email; fetch(ENDPOINT, { method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify(payload) }).catch(function(){}); } var meta = (window.ShopifyAnalytics && window.ShopifyAnalytics.meta) || {}; track('viewed_page', { url: location.pathname, page_title: document.title }); if (meta.product) track('viewed_product', { product_id: String(meta.product.id), product_title: meta.product.title, url: location.pathname }); if (meta.article) track('viewed_blog', { article_id: String(meta.article.id||''), article_title: meta.article.title||document.title, blog_title: (meta.blog&&meta.blog.title)||'', url: location.pathname }); if (meta.collection) track('viewed_collection', { collection_id: String(meta.collection.id), collection_title: meta.collection.title, url: location.pathname }); var sq = new URLSearchParams(location.search).get('q'); if (sq && location.pathname.indexOf('/search') !== -1) track('searched', { query: sq, url: location.pathname }); if (location.pathname === '/cart') track('viewed_cart', { url: location.pathname }); })(); </script>

Tracks: product views, blog reads, searches, cart visits, page views. Customers are auto-identified when logged in (via Liquid) or when clicking from an email link (via ?meh_e=EMAIL).
